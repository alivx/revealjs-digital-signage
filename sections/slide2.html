httpGet()
            {
              case "$configuredClient" in
                curl)  curl -A curl -s "$@" ;;
                wget)  wget -qO- "$@" ;;
                httpie) http -b GET "$@" ;;
                fetch) fetch -q "$@" ;;
              esac
            }
